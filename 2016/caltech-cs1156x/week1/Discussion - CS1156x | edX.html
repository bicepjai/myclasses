              (Apologies for not providing just a link to this solution. Not certain where else people will have easy access.)

              Here is a "bare bones" MATLAB implementation for Questions 7-10. There are four functions (each in its own file) and a main script (to preserve the workspace after execution). The functions are more general than required by the main script but I have retained the extra functionality in case it is needed in future homework. For N=100, the PLA fails to converge on approximately one in five thousand runs. Given this possibility, the PLA loop has a second condition to ensure a tidy termination. The various "docstrings" have been moved into a summary that precedes the code. An occasional usage note has been added to these comments.

              ==========================
HOMEWORK 1 - 2D Perceptron
==========================

x = randpts (n [,b]) 
    returns a 3xn array of uniformly distributed random 2D points from the space 
    [-b,+b] x [-b,+b]. Each column in x refers to one 2D point and has the form: 
    [x0; x1; x2] where x0=1. If b is omitted, b=1 is assumed.

[w,p] = randline ([b]) 
    returns a random line in the space [-b,+b] x [-b,+b] and two points on this line. 
    The line is specified by w, a vector of weights of the form [intercept; gradient; -1]. 
    The points are stored in a 3x2 array with each column representing a point using the 
    form [x0; x1; x2] where x0=1. If b is omitted, b=1 is assumed. (Point p is not used
    in the homework script.) 

[x,y,w] = SeparableData (n) 
    returns a random line in the space [-1,+1] x [-1,+1] and n data points classified 
    according to which side of the line they are on. Classifications are returned in y,
    a 1xn vector of values from {-1,+1}. The n data points are returned in x, a 3xn array
    where each column represents a 2D point and has the form [x0; x1; x2] where x0=1. 
    The line is returned as a column vector w of three weights (see randline). However,
    this target function is supposed to be unknown, so it is not assigned a variable
    in the homework script's workspace.  

[w,k] = prcptrn (iw,x,y [,maxi]);
    runs the perceptron learning algorithm with initial weights iw on training points 
    x and classifications y, and returns the weights w that separate the data. Also 
    returns k, the number of iterations required to learn w. If maxi, the maximum number 
    of iterations is not supplied, 1e4 is assumed.

HW1 (script)
    For R runs, creates N training points and classifications, and M external points
    and classifications, runs the perceptron learning algorithm on the training data
    to learn a hypothesis that separates this data and then classifies the external 
    points using this hypothesis. After R runs, reports the average number of PLA 
    iterations required to converge to a hypothesis and the average probability of
    external point misclassification.

==========================

function x = randpts (n,b)
  if nargin<2, b=1; end                      % default input space [-1,1] x [-1,1]
  x = [ones(1,n); 2*b*rand(2,n)-b];          % 3xn random points of form [x0;x1;x2], x0=1
end

function [w,p] = randline (b)
  if nargin<1, b=1; end                      % default space [-1,+1] x [-1,+1]
  p = randpts(2,b);                          % two random points in input space
  m = (p(3,2)-p(3,1))/(p(2,2)-p(2,1));       % gradient of line between these points
  c = p(3,1) - m*p(2,1);                     % y-intercept of this line 
  w = [c; m; -1];                            % line specification as weights
end

function [x,y,wf] = SeparableData (n)
  wf = randline;                             % generate target function in [-1,1] x [-1,1]
  x = randpts(n);                            % generate n random data points
  y = sign(wf'*x);                           % classify according to target function
end

function [w,k] = prcptrn (iw,x,y,maxi)
  if nargin<4, maxi = 1e4; end               % default most iterations is 1e4
  bad = @(w,x,y) find(sign(w'*x)~=y);        % return misclassified points

  k = 0; w = iw;                             % initially, use the supplied weights
  m = bad(w,x,y);                            % identify misclassified points

  while ~isempty(m) && k<=maxi
    p = randi(length(m));                    % choose a random misclassified point
    w = w + y(m(p))*x(:,m(p));               % adjust weights based on that point  
    m = bad(w,x,y);                          % identify misclassified points
    k = k+1;                                 % increment iterations required
  end
end

==========================

N = 100;                                     % number of training points
M = 10*N;                                    % number of external points
R = 1000;                                    % number of runs

ReqdIters = zeros(1,R);                      % record of iterations required
ProbError = zeros(1,R);                      % record of error probability

for run=1:R
  [X,Y] = SeparableData (N+M);               % data points, classifications
  xt = X(:,1:N);  xe = X(:,N+1:end);         % xt, xe = training, external data points 
  yt = Y(1:N);    ye = Y(N+1:end);           % yt, ye = training, external classifications

  [w,k] = prcptrn ([0;0;0],xt,yt);           % learn weights based on training data
  ReqdIters(run) = k;                        % record iterations required

  g = sign(w'*xe);                           % predict external classifications
  ProbError(run) = sum(g~=ye)/M;             % record relative frequency of error
end

fprintf ('2D Perceptron Example\n\n');
fprintf ('Number of training points:              %8.0d\n', N);
fprintf ('Number of external points:              %8.0d\n', M);
fprintf ('Number of runs:                         %8.0d\n', R);
fprintf ('Mean number of iterations to converge:  %8.0f\n', mean(ReqdIters));
fprintf ('Mean probability of misclassification:  %8.5f\n', mean(ProbError));

==========================
END OF DOCUMENT

                My solution in C:

                #include <stdio.h>
#include <stdlib.h>

#define N 10

typedef struct point
{
    float x,y;
} point;

typedef struct lin_funct
{
    point p1,p2;
} lin_funct;

float func_eval(lin_funct f, float x)
{
    float slope = (f.p2.y - f.p1.y) / (f.p2.x - f.p1.x);

    return slope*x + (-slope*f.p1.x) + f.p1.y;
}

int sign(float val)
{
    return (0.0f < val) - (val < 0.0f);
}

int main()
{

    int i,run,runs = 100000,total_iterations = 0,num_sep_points = 1000,disagreements = 0;

    for (run = 0; run <runs; run++)
    {
        point data_set[N];
        float weights[3];
        int desired_value[N];

        /// Generating random target function
        lin_funct target_function;
        target_function.p1.x = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;
        target_function.p1.y = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;
        target_function.p2.x = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;
        target_function.p2.y = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;

        /// Generating random points data set
        for (i = 0; i < N; i++)
        {
            data_set[i].x = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;
            data_set[i].y = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;
        }

        /// Evaluating if the generated points are above or below the target function
        for (i = 0; i < N; i++)
        {
            if (data_set[i].y > func_eval(target_function,data_set[i].x))
                desired_value[i] = 1;
            else
                desired_value[i] = -1;

        }

        /// Initializing the weight vector
        weights[0] = 0.0f;
        weights[1] = 0.0f;
        weights[2] = 0.0f;

        int error,iteration = 0;
        do
        {
            int perceptron_eval;
            iteration++;


            /// finding a random misclassified point
            do
            {
                i = rand()%N;
                perceptron_eval = sign(weights[0] + weights[1]*data_set[i].x + weights[2]*data_set[i].y);
            }
            while(desired_value[i] == perceptron_eval);

            /// update the weights
            weights[0] = weights[0] + (desired_value[i] - perceptron_eval);
            weights[1] = weights[1] + (desired_value[i] - perceptron_eval)*data_set[i].x;
            weights[2] = weights[2] + (desired_value[i] - perceptron_eval)*data_set[i].y;;

            /// Evaluate total error
            error = 0;
            for (i = 0; i < N; i++)
            {
                perceptron_eval = sign(weights[0] + weights[1]*data_set[i].x + weights[2]*data_set[i].y);
                if (desired_value[i] != perceptron_eval)
                    error++;
            }
        }
        while(error != 0); /// while not converged
        total_iterations += iteration;


        point separate_point;
        for(i = 0; i < num_sep_points; i++)
        {
            separate_point.x = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;
            separate_point.y = -1 + ((float)rand()/(float)(RAND_MAX)) * 2.0f;

            int separate_desired;

            if (separate_point.y > func_eval(target_function,separate_point.x))
                separate_desired = 1;
            else
                separate_desired = -1;

            int perceptron_eval = sign(weights[0] + weights[1]*separate_point.x + weights[2]*separate_point.y);

            if (separate_desired != perceptron_eval)
                disagreements++;

        }
    }
    printf("For N: %d\n",N);
    printf("the average convergence iteration is %f\n", (float)total_iterations/(float)runs);
    printf("Probability of f(x) != g(x) %f\n", (float)disagreements/(float)(num_sep_points*runs));

    return 0;
}

                  #!/usr/bin/python

import numpy as np
import pytest

from itertools import count

NUM_RUNS = 1000
MONTE_CARLO_NUM_POINTS = 400


def main():
    for N in (10, 100):  # N is the number of sample points
        results = [run_experiment(N) for _ in range(NUM_RUNS)]
        num_iterations, p_f_neq_g = np.mean(results, axis=0)
        print('N={}: iter={}, prob={}'.format(N, num_iterations, p_f_neq_g))


def run_experiment(N):
    """
    Runs an experiment with `N` random points.
    """
    f = random_line()
    sample_points = random_points(N)
    correct_labels = label_points(sample_points, f)
    g, num_iterations = perceptron_learning(sample_points, correct_labels)
    p_f_neq_g = calculate_p_f_neq_g(f, g)
    return num_iterations, p_f_neq_g


def calculate_p_f_neq_g(f, g, points=None):
    """
    Calculate the probability P(f(x) != g(x)) using Monte Carlo method.
    `f` and `g` are 3-element arrays [w0, w1, w2] representing a line:
    w0 + w1 * x + w2 * y = 0.
    `points` are used when doing Monte Carlo; if not specified, they are
    generated randomly.
    """
    if points is None:
        points = random_points(MONTE_CARLO_NUM_POINTS)
    label_f = label_points(points, f)
    label_g = label_points(points, g)
    return np.mean(~np.isclose(label_f, label_g))


def perceptron_learning(points, correct_labels):
    w = np.zeros(3)
    for num_iterations in count(start=0, step=1):
        pred_labels = label_points(points, w)
        miss_condition = ~np.isclose(pred_labels, correct_labels)

        # If all points are correctly classified
        if not miss_condition.any():
            break

        miss_point_indexes = miss_condition.nonzero()[0]
        miss_point_index = np.random.choice(miss_point_indexes, 1)[0]
        miss_point = points[miss_point_index]
        miss_point_correct_label = correct_labels[miss_point_index]
        x = np.hstack((1, miss_point))
        w += miss_point_correct_label * x

    return w, num_iterations


def label_points(points, line):
    """
    Labels a list of points using a line. A label can be either -1, 0, or 1.
    Returns a list of labels corresponding to the points.
    A list of points is represented by a 2-dimensional array `points`.
    A line: w0 + w1 * x + w2 * y = 0 is represented by
    an array `line` [w0, w1, w2].
    """
    num_points = points.shape[0]
    a = np.column_stack((np.ones(num_points), points))
    # sign( [1., x, y] . [w0, w1, w2] )
    return np.sign(a.dot(line))


def find_line(point_1, point_2):
    """
    Finds the line crossing `point_1` and `point_2`. Returns an array
    [w0, w1, w2] representing a line: w0 + w1 * x + w2 * y = 0.
    `point_1` and `point_2` are 2-element arrays.
    """
    # [x1 y1] . [w1] = [-w0] ==> a . x = b
    # [x2 y2]   [w2]   [-w0]
    a = [point_1, point_2]
    w0 = 1.
    b = [-w0, -w0]
    w1, w2 = np.linalg.solve(a, b)
    return np.array([w0, w1, w2])


def random_line():
    point_1, point_2 = random_points(2)
    return find_line(point_1, point_2)


def random_points(N):
    return np.random.uniform(-1., 1., size=(N, 2))

                              import random
import numpy as np

# globals:
vectors = [0, 0]  # to be used in function proportion_of_disagreements().


def make_target_f(n):
    """Creates a target function f ( a straight line), and use it to assign
     classes (sign) to randomly generated points in uniform 2D [-1, 1]x[-1, 1] plane.
    :param n: number of training points to create.
    :return: list of points where each point is a tuple(x1, y1, sign).
    """
    # make a line with two points in 2D  (target function f):
    x1, y1 = random.uniform(-1, 1), random.uniform(-1, 1)
    x2, y2 = random.uniform(-1, 1), random.uniform(-1, 1)

    # using equation of a line with two points x and y in 2D calculate weights:
    w0 = -(y2-y1)*x1 + (x2 - x1)*y1
    w1 = y2 - y1
    w2 = - (x2 - x1)
    w = np.array([w0, w1, w2])

    vectors[0] = w  # save vector of weights 

    points = []

    for _ in range(n):
        point = random.uniform(-1, 1), random.uniform(-1, 1)
        x = np.array([1, point[0], point[1]])

        if np.sign((w.transpose()).dot(x)) > 0:
            points.append((point[0], point[1], 1))
        else:
            points.append((point[0], point[1], -1))

    return points


def perceptron(points):
    """Runs PLA to find a function g which correctly classifies all points if
    points are linearly separable.
    :param points: list of points created by make_target_f().
    :return: number of iteration it took to converge.
    """
    count = 0
    w = np.array([0, 0, 0])
    misclassified = 1

    while misclassified:
        count += 1
        misclassified = []
        for point in points:
            x = np.array([1, point[0], point[1]])
            if np.sign((w.transpose()).dot(x)) != point[2]:
                misclassified.append(point)

        if misclassified:
            point = random.choice(misclassified)
            w0 = w[0] + point[2]
            w1 = w[1] + point[0] * point[2]
            w2 = w[2] + point[1] * point[2]
            w = np.array([w0, w1, w2])

    vectors[1] = w  # save vector of weights

    return count


def num_iterations(N, n):
    """Finds approximate number of iterations for PLA to converge.
    :param N: number of points.
    :param n: number of simulations to run
    :return: number of iterations averaged over n simulations.    
    """
    counts = []
    for _ in range(n):
        points = make_target_f(N)
        iterations = perceptron(points)
        counts.append(iterations)

    return sum(counts)/float(n)


def proportion_of_disagreements(N, n):
    """Finds approximate proportion of disagreements between f and g.
    :param N: number of points to run f and g.
    :param n: number of simulations to run.
    :return: approximate proportion of disagreements (prob. to be misclassified by g).
    """
    prob_list = []

    for _ in range(n):
        points = make_target_f(N)
        perceptron(points)

        count = 0
        for dummy_i in range(1000): 
            point = random.uniform(-1, 1), random.uniform(-1, 1)
            x = np.array([1, point[0], point[1]])
            if np.sign((vectors[0].transpose()).dot(x)) != np.sign((vectors[1].transpose()).dot(x)):
                count += 1

        prob_list.append(count/1000.0)

    return sum(prob_list)/float(n)
