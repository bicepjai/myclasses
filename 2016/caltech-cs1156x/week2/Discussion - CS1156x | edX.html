        import numpy as np
import time

def linear_reg_experiment(N,n,noise_ratio):  # N is population size, n is sample size

    # generate random samples using a uniform distribution from -1 to +1
    P_N=np.array([np.random.uniform(-1,+1,N+n),np.random.uniform(-1,+1,N+n)])
    X_n=np.array(P_N[:n,:n])
    P_N=P_N[:,n:N+n]

    # add x0, add transformed features
    X=np.ones(shape=(n,6))
    P=np.ones(shape=(N,6))
    X[:,1:3]=X_n.T
    X[:,3]=X[:,1]*X[:,2]
    X[:,4]=X[:,1]**2
    X[:,5]=X[:,2]**2
    P[:,1:3]=P_N.T
    P[:,3]=P[:,1]*P[:,2]
    P[:,4]=P[:,1]**2
    P[:,5]=P[:,2]**2

    # classify training and test sets
    y=np.sign(X[:,1]**2+X[:,2]**2-0.6)
    y_N=np.sign(P[:,1]**2+P[:,2]**2-0.6)

    # add noise into training set
    if 1==1:      # activate this line if you want to add noise to the training set
        noise_index=np.arange(0,n)
        flip=int(noise_ratio*n)
        np.random.shuffle(noise_index)
        flip_index=noise_index[:flip]
        y[flip_index]=-y[flip_index]

    # add noise into training set
    if 1==1:      # activate this line if you want to add noise to the test set
        noise_N_index=np.arange(0,N)
        flip_N=int(noise_ratio*N)
        np.random.shuffle(noise_N_index)
        flip_N_index=noise_N_index[:flip_N]
        y_N[flip_N_index]=-y_N[flip_N_index]

    # normal equations
    X_dagger=np.dot(np.linalg.pinv(np.dot(X.T,X)),X.T)
    w=np.dot(X_dagger,y)

    # classify training set using w
    h=np.array(np.sign(np.dot(X,w)))
    error_mask=h!=y

    # classify test set using w
    h_N=np.array(np.sign(np.dot(P,w)))
    error_mask_N=h_N!=y_N

    return np.count_nonzero(error_mask)/len(error_mask),np.count_nonzero(error_mask_N)/len(error_mask_N),w

          Here's my simulator for Problem 1.

          The code is in Python, fully vectorized and runs in under 5 seconds on my (2011 dated) machine. 

          import numpy as np

def HW2_problem1_simplified_do_the_sim(m_flips=10, n_coins= 1000, k_trials = 100000):
    """
    This is the workhorse for problems 1 and 2
    it returns (result_first_vector_flipped, result_random_selection, result_minimum)        
    """
    A = np.random.randint(0,2, size= (m_flips, n_coins, k_trials), dtype=np.bool)
    #  A = np.random.randint(0,2, size= (m_flips, n_coins, k_trials), dtype=np.int32)
    # if you instead comment out the first line, and uncomment the 2nd line, you get much slower performance    

    results_A = np.sum(A, axis =0, dtype=np.int32)

    initial_coin_vec = results_A[0,:]    
    my_new_arrangement = np.arange(k_trials)    

    random_indices = np.random.randint(0, n_coins, size = k_trials)    
    random_coin_vec = results_A[random_indices, my_new_arrangement] # tricky

    min_at_any_point_in_time_vec = np.min(results_A,axis=0)

    return initial_coin_vec/m_flips, random_coin_vec/m_flips, min_at_any_point_in_time_vec/m_flips

            from pylab import *
from random import choice

num_tosses = 10
num_coins = 1000
num_iters = 100000
v1 = []
vrand = []
vmin = []

for i in arange(num_iters):
    # Toss Coins
    coins = np.random.binomial(n=1,p=0.5,size=(num_coins,num_tosses))
    # Get 3 Coin Results
    vmin.append(np.min(coins.sum(axis=1))/float(num_tosses))
    v1.append(coins[0].sum()/float(num_tosses))
    vrand.append(choice(coins).sum()/float(num_tosses))

print 'v1 mean: ', np.mean(v1)
print 'vrand mean: ', np.mean(vrand)

                import Jama.Matrix;

                public class LinearRegression { public static void main(String[] args) {

                    double sumOfErrorsInSample = 0;
    double sumOfErrorsOutOfSample = 0;
    int runs = 1000;
    for (int j = 0; j < runs; j++) {

        //Training step
        int numberOfTrainingPoints = 100;
        Problem problem = new Problem(numberOfTrainingPoints);
        Point[] xn = problem.getXn();
        int[] yn = problem.getYn();
        Classifier g = linearRegression(problem.getN(), xn, yn, 2);
        int errors = 0;
        for (int i = 0; i < xn.length; i++) {
            Sign h = h(g.w, xn[i].x, xn[i].y);

            if (!h.equals(Sign.sign(yn[i]))) {
                errors++;
            }
        }

        sumOfErrorsInSample += (double) errors / numberOfTrainingPoints;

        // Testing step
        errors = 0;
        int numberOfTestPoints = 1000;
        for (int k = 0; k < numberOfTestPoints; k++) {
            Point test = Point.randomPoint();
            double result = Problem.f(problem.getLineAngularCoefficient(), problem.getLinePointX(),
                    problem.getLinePointY(), test);
            Sign expected = Sign.sign(result);
            Sign answer = h(g.w, test.x, test.y);
            if (!expected.equals(answer)) {
                errors++;
            }
        }

        sumOfErrorsOutOfSample += (double) errors / numberOfTestPoints;

        System.out.println("Vetor w = [" + g.w[0] + ", " + g.w[1] + ", " + g.w[2] + "]");
    }

    System.out.println("Average E_In = " + (double) sumOfErrorsInSample / runs);
    System.out.println("Average E_Out = " + (double) sumOfErrorsOutOfSample / runs);
}

static Sign h(double[] w, double x, double y) {
    double sum = w[0] + w[1] * x + w[2] * y;
    Sign h = Sign.sign(sum);
    return h;
}

public static Classifier linearRegression(int n, Point[] xn, int[] yn, int dim) {
    int d = dim + 1;
    double[][] X = new double[n][d];
    double[] Y = new double[n];

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < d; j++) {

            if (j == 0) {
                X[i][j] = 1;
            } else if (j == 1) {
                X[i][j] = xn[i].x;
            } else if (j == 2) {
                X[i][j] = xn[i].y;
            }
        }

        Y[i] = yn[i];
    }

    Matrix matrixX = Matrix.constructWithCopy(X);
    Matrix transpX = matrixX.transpose();
    Matrix pseudoInverseX = transpX.times(matrixX).inverse().times(transpX);

    double[] ynDouble = new double[yn.length];
    for (int i = 0; i < yn.length; i++) {
        ynDouble[i] = yn[i];
    }
    Matrix matrixY = new Matrix(ynDouble, n);
    Matrix wlin = pseudoInverseX.times(matrixY);
    double[] vectorWLin = wlin.getRowPackedCopy();

    return new Classifier(vectorWLin, 1);
}

                }

                      There are two questions in Homework 2 that do not require coding. Here is my reasoning for these questions. Alerts to misconceptions and other errors gratefully received. Should I move these responses to their respective homework discussion threads?

                      Question 3.

                      Given:

                      P(f(x))=1−μP(¬f(x))=μP(y|y=f(x))=λP(y|y≠f(x))=1−λ

                      the probability that h makes an error in approximating y is the probability that:

                      h correctly chooses f(x) but y≠f(x); or
                      h errs in choosing f(x) and y=f(x).

                      This is:

                      P(μ,λ)=P(f(x)).P(y|y≠f(x))+P(¬f(x)).P(y|y=f(x))=(1−μ).(1−λ)+μ.λ

                      Question 4.

                      Given P(μ,λ)=(1−μ)(1−λ)+μλ, P(μ,λ) is independent of μ when ∂∂μP(μ,λ)=0. Thus,

                      ∂∂μP(μ,λ)=∂∂μ((1−μ)(1−λ)+μλ)=∂∂μ(1−μ−λ+2μλ)=0−1−0+2λ=0

                      whereupon λ=12, and P(μ,12)=(1−μ)(1−12)+μ12=12−12μ+12μ=12, independent of μ.

                      2. @LeeLafferty: Looks good! Thanks for writing this up. This problem is short enough that I think we can try keeping it in this thread and redirecting people here for now.

                        One additional note is that you don't need to take the μ derivative at all - you can just rewrite

                        P(μ,λ)=1−λ+μ(1−2λ)

                        and the μ term drops out when λ=1/2, that is, the probability of error is 1/2 regardless of μ.

                        An interpretation of this is that when the output y is completely uncorrelated with f, it doesn't matter how good or bad your hypothesis is at approximating f. Alternatively, λ=1/2 corresponds to completely random data or pure noise, which violates "Axiom 1" from Lecture 1, "a pattern exists".

while (r < runs) {
    r <- r+1
    s <- rbinom(coins,tosses,p) # list of successes 
    c1 <- s[1] # number of successes for first coin
    cr <- s[sample(1:coins,1)] # number of successes for random coin
    cmin <- s[match(min(s), s)] # number of successes for coin with minimum number of heads

    # v1 is the fraction of heads for coin 1 (c1) out of 10 tosses
    v1 = append(v1, c1/tosses)

    # vrand is the fraction of heads for the random coin (crand) out of 10 tosses
    vr = append(vr, cr/tosses)

    # vmin is the fraction of heads for the coin with minimum number of heads (cmin) out of 10 tosses
    vm = append(vm, cmin/tosses)

    }

print(paste0("average V1: ", sum(v1) / runs))

                                  HW2 problem 3:

                                  When we use h to approximate the noisy version of f, we have four possible situations for the correctness of the result. I summarized them in the next table:

                                  y=f(x)y≠f(x)h≈fCorrectIncorrecth≉fIncorrectCorrect

                                  According to the table we get an incorrect result when this event happens (I call it A):

                                  A=[(h≈f) AND (y≠f(x)] OR [(h≉f) AND (y=f(x)]

                                  Now we can calculate the probability of A:

                                  P(A)=(1−μ)(1−λ)+μλ=1−λ−μ+2μλ

                                  In the previous one we notice that when −μ+2μλ=0 the probability is independent of μ, because it disappears in the equation, and this implies λ=12.
